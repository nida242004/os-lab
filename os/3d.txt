#include<stdio.h>
#include<string.h>

void main(){
	char fn[10],pat[10],temp[200];
	FILE *fp;
	printf("Enter the filename\n");
	scanf("%s",fn);
	printf("ENter the pattern to be searched\n");
	scanf("%s",pat);
	fp = fopen(fn,"r");
	while(!feof(fp)){
		fgets(temp,100,fp);
		if(strstr(temp,pat))
			printf("%s",temp);
	}
	fclose(fp);
}
